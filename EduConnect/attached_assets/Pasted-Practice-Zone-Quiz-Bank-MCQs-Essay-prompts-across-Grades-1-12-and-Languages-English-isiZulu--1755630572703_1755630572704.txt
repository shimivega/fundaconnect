2. Read Aloud on Word Selection (Web Speech API)

Here’s a React component that lets learners click or highlight a word, and the browser will speak it aloud:

'use client'
import { useEffect } from 'react'

export default function Reader() {
  useEffect(() => {
    function handleSelection() {
      const selection = window.getSelection()?.toString().trim()
      if (selection) {
        const utterance = new SpeechSynthesisUtterance(selection)
        utterance.lang = 'en-US' // can set dynamically: 'zu-ZA' for isiZulu, 'af-ZA' for Afrikaans
        window.speechSynthesis.speak(utterance)
      }
    }

    document.addEventListener('mouseup', handleSelection)
    return () => {
      document.removeEventListener('mouseup', handleSelection)
    }
  }, [])

  return (
    <div className="p-4 border rounded-xl">
      <h2 className="font-bold text-lg mb-2">Practice Reading</h2>
      <p className="text-gray-700">
        Select any word in this paragraph and I will read it aloud. Learners can use this feature to
        practice pronunciation and improve reading fluency in different languages.
      </p>
    </div>
  )
}
